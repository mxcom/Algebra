\begin{bmatrix}
2 & -4 & 3 & -1 & | & 2\\

\begin{bmatrix}

0 & 0 & -3.5 & 3.5 & | & 14\\
0 & 0 & 3.5 & a & | & c\\
\end{bmatrix}
\xrightarrow{\text{III - II}}
\begin{bmatrix}
2 & -4 & 3 & -1 & | & 2\\
0 & 0 & -3.5 & 3.5 & | & 14\\

\end{bmatrix}
