\sysdelim..\systeme{
2x + 4y + z = 7,
4x + 2y + 3z = 8,
x + 5y + 2z = 0
}\\~\\

$
\begin{bmatrix}
2 & 4 & 1 & | & 7\\
4 & 2 & 3 & | & 8\\
1 & 5 & 2 & | & 0
\end{bmatrix}
\xrightarrow{\text{II - (-2} \cdot \text{I)}}
\begin{bmatrix}
2 & 4 & 1 & | & 7\\
0 & -6 & 1 & | & -6\\
1 & 5 & 2 & | & 0
\end{bmatrix}
\xrightarrow{\text{III - (0.5} \cdot \text{I)}}
\begin{bmatrix}
2 & 4 & 1 & | & 7\\
0 & -6 & 1 & | & -6\\
0 & 3 & 1.5 & | & -3.5
\end{bmatrix}
\xrightarrow{\text{III - (-0.5} \cdot \text{II)}}\\~\\
\begin{bmatrix}
2 & 4 & 1 & | & 7\\
0 & -6 & 1 & | & -6\\
0 & 0 & 2 & | & -6.5
\end{bmatrix}
\longrightarrow
\sysdelim..\systeme{
2x + 4y + z = 7,
-6y + 1z = -6,
2z = -6.5
}
\longrightarrow
z = \frac{-13}{4}, y = \frac{11}{24}, x = \frac{101}{24}
$\\

\sysdelim..\systeme{
x_1 + x_2 + x_3 + x_4 = 7,
x_1 + 2x_2 + 3x_3 - x_4 = 6,
2x_1 - 3x_2 - 2x_3 + 4x_4 = 7,
3x_1 - x_2 + x_3 + 2x_4 = 10
}\\~\\

$
\begin{bmatrix}
1 & 1 & 1 & 1 & | & 7\\
1 & 2 & 3 & -1 & | & 6\\
2 & -3 & -2 & 4 & | & 7\\
3 & -1 & 1 & 2 & | & 10
\end{bmatrix}
\xrightarrow{\text{II - I}}
\begin{bmatrix}
1 & 1 & 1 & 1 & | & 7\\
0 & 1 & 2 & -2 & | & -1\\
2 & -3 & -2 & 4 & | & 7\\
3 & -1 & 1 & 2 & | & 10
\end{bmatrix}
\xrightarrow{\text{III - (2} \cdot \text{I)}}\\~\\~\\
\begin{bmatrix}
1 & 1 & 1 & 1 & | & 7\\
0 & 1 & 2 & -2 & | & -1\\
0 & -5 & -4 & 2 & | & -7\\
3 & -1 & 1 & 2 & | & 10
\end{bmatrix}
\xrightarrow{\text{IV - (3} \cdot \text{I)}}
\begin{bmatrix}
1 & 1 & 1 & 1 & | & 7\\
0 & 1 & 2 & -2 & | & -1\\
0 & -5 & -4 & 2 & | & -7\\
0 & -4 & -2 & -1 & | & -11
\end{bmatrix}
\xrightarrow{\text{III - (5} \cdot \text{II)}}\\~\\~\\
\begin{bmatrix}
1 & 1 & 1 & 1 & | & 7\\
0 & 1 & 2 & -2 & | & -1\\
0 & 0 & 6 & -8 & | & -12\\
0 & -4 & -2 & -1 & | & -11
\end{bmatrix}
\xrightarrow{\text{IV - (4} \cdot \text{II)}}
\begin{bmatrix}
1 & 1 & 1 & 1 & | & 7\\
0 & 1 & 2 & -2 & | & -1\\
0 & 0 & 6 & -8 & | & -12\\
0 & 0 & 6 & -9 & | & -15
\end{bmatrix}
\xrightarrow{\text{IV - III}}\\~\\~\\
\begin{bmatrix}
1 & 1 & 1 & 1 & | & 7\\
0 & 1 & 2 & -2 & | & -1\\
0 & 0 & 6 & -8 & | & -12\\
0 & 0 & 0 & -1 & | & -3
\end{bmatrix}
\longrightarrow
\sysdelim..\systeme{
x_1 + x_2 + x_3 + x_4 = 7,
x_2 + 2x_3 - 2x_4 = -1,
6x_3 - 8x_4 = -12,
-1x_4 = -3
}
\longrightarrow
$
